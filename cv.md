```
    if len(arr) == sum(x for x in arr if type(x) != type('x')):
```

```
def anagrams(word, words):
    res = []
    word = ''.join(sorted(list(word)))
    for i in words:
        wordb = ''.join(sorted(list(i)))
        if wordb == word:
            res.append(i)
    return res
```
